\chapter{KẾT LUẬN}

\section{Tóm tắt đề tài}

Trong bối cảnh các hệ thống khuyến nghị tập trung (Cloud-Centric) đang bộc lộ nhiều điểm yếu chí mạng về nguy cơ rò rỉ quyền riêng tư, độ trễ hệ thống và tiêu tốn băng thông mạng, đề tài này đã đặt ra mục tiêu nghiên cứu và giải quyết bài toán cân bằng giữa chất lượng khuyến nghị và bảo vệ dữ liệu người dùng. 

Giải pháp cốt lõi mà nhóm nghiên cứu đề xuất và chứng minh thành công là việc áp dụng kiến trúc \textbf{Device-Cloud Collaborative Learning (DCCL)} kết hợp cùng \textbf{Mạng nơ-ron đồ thị (Graph Neural Networks - GNN)}. Bằng việc phân tách quy trình học: để Đám mây (Cloud) chịu trách nhiệm học tri thức toàn cục (Global Knowledge) và Thiết bị (Device) đảm nhận suy luận đồ thị con cục bộ (Local Subgraph Inference) để cá nhân hóa, hệ thống đã giải quyết trọn vẹn các thách thức kể trên mà vẫn duy trì độ chính xác ấn tượng so với các mô hình truyền thống.

\section{Tổng hợp các công việc đã thực hiện}

Xuyên suốt quá trình nghiên cứu và phát triển, nhóm đã hoàn thành một quy trình toàn vẹn (end-to-end pipeline) bao gồm các công việc chính sau:

\begin{itemize}
    \item \textbf{Khảo sát và hệ thống hóa lý thuyết:} Nghiên cứu chuyên sâu về nền tảng của Hệ thống khuyến nghị, mạng GNN (đặc biệt là thuật toán GraphSAGE) và cơ chế học phối hợp DCCL thông qua các công trình khoa học uy tín.
    
    \item \textbf{Tiền xử lý và mô hình hóa dữ liệu:} 
    \begin{itemize}
        \item Xử lý tập dữ liệu thực tế \textbf{Amazon Product Data (nhóm hàng Software - 5-core)}.
        \item Sử dụng các mô hình ngôn ngữ lớn (Sentence Transformers) để trích xuất đặc trưng văn bản.
        \item Xây dựng thành công Đồ thị Dị nhất (Heterogeneous Graph) mô phỏng chính xác các mối quan hệ phức tạp (Review, Also\_buy) giữa Người dùng và Sản phẩm.
    \end{itemize}
    
    \item \textbf{Phát triển hệ thống Đám mây (Cloud Server):}
    \begin{itemize}
        \item Thiết kế và huấn luyện mô hình 2-layer GraphSAGE để sinh ra các vector nhúng sản phẩm (Global Item Embeddings) chất lượng cao.
        \item Xây dựng RESTful API bằng framework \textbf{FastAPI} để đóng vai trò là Global Server, phục vụ việc truy vấn vector nhúng với độ trễ thấp.
    \end{itemize}
    
    \item \textbf{Phát triển luồng Suy luận Thiết bị (Device Inference):}
    \begin{itemize}
        \item Chuyển đổi thành công bộ tổng hợp SAGE (SAGE Aggregator) và bộ giải mã cạnh (DotProduct Decoder) sang định dạng \textbf{TensorFlow Lite (.tflite)}.
        \item Hiện thực luồng xử lý cục bộ: Tải vector nhúng từ Cloud $\rightarrow$ Tổng hợp đặc trưng láng giềng $\rightarrow$ Xếp hạng (Ranking) trực tiếp trên thiết bị để đưa ra danh sách khuyến nghị.
    \end{itemize}
    
    \item \textbf{Đánh giá Thực nghiệm toàn diện:} Chạy các kịch bản kiểm thử nghiêm ngặt, so sánh trực tiếp GraphSAGE với GCN. Kết quả thực nghiệm đã khẳng định tính khả thi của DCCL khi GraphSAGE duy trì mức AUROC xuất sắc (>0.93) với độ sụt giảm quy nạp (Inductive Gap) vô cùng nhỏ khi triển khai trên thiết bị mới.
\end{itemize}

\section{Ưu điểm và hạn chế}

\subsection{Ưu điểm}
Giải pháp do nhóm đề xuất mang lại những cải tiến vượt trội so với các hệ thống truyền thống:
\begin{itemize}
    \item \textbf{Bảo mật quyền riêng tư (Privacy-Preserving):} Mọi dữ liệu hành vi thô (raw click logs) của người dùng đều được giữ lại hoàn toàn trên thiết bị. Việc tính toán vector nhúng người dùng được thực thi cục bộ.
    
    \item \textbf{Tiết kiệm băng thông (Bandwidth Efficiency):} Hệ thống loại bỏ việc phải liên tục upload dữ liệu hành vi. Thiết bị chỉ cần tải (download) danh sách các vector nhúng sản phẩm có kích thước cực nhỏ và được lưu trữ tĩnh.
    
    \item \textbf{Độ trễ bằng 0 ở khâu xếp hạng (Zero-Latency Ranking):} Do quá trình nhân ma trận tính điểm số được mô hình TFLite xử lý ngay trên vi xử lý của thiết bị di động, người dùng có thể nhận được kết quả khuyến nghị ngay lập tức.
    
    \item \textbf{Sức mạnh Quy nạp (Inductive Generalization):} Nhờ GraphSAGE học "hàm tổng hợp" thay vì học thuộc lòng vector, hệ thống xử lý hoàn hảo vấn đề người dùng mới tải ứng dụng (Unseen Users).
\end{itemize}

\subsection{Hạn chế}
Bên cạnh những kết quả đạt được, đồ án vẫn còn một số điểm giới hạn cần được nhìn nhận khách quan:
\begin{itemize}
    \item \textbf{Môi trường thực nghiệm mô phỏng:} Quá trình kiểm thử phía Client hiện mới chỉ dừng lại ở việc dùng Python Script giả lập môi trường di động. Nhóm chưa có điều kiện triển khai native thành một file APK/IPA thực tế trên smartphone để đo lường cụ thể mức độ ngốn RAM và tiêu hao pin.
    
    \item \textbf{Vấn đề Cold-Start đối với Sản phẩm mới:} Mô hình giải quyết rất tốt bài toán người dùng mới, nhưng nếu trên hệ thống đám mây xuất hiện một sản phẩm hoàn toàn mới (chưa có embedding), thiết bị sẽ không thể tính toán xếp hạng cho sản phẩm đó cho đến khi Cloud huấn luyện lại mô hình.
    
    \item \textbf{Luồng cập nhật tĩnh:} Hệ thống hiện tại chủ yếu theo hướng từ Cloud đẩy xuống Device. Chưa có một quy trình tự động và bảo mật (Federated Learning) để thiết bị đẩy các bản cập nhật Gradient ngược lên Cloud giúp mô hình toàn cục ngày càng thông minh hơn.
\end{itemize}

\section{Hướng phát triển trong tương lai}

Dựa trên nền tảng kỹ thuật vững chắc đã xây dựng, cùng việc phân tích những hạn chế hiện đọng, nhóm đề xuất các hướng phát triển tiếp theo nhằm hoàn thiện và thương mại hóa hệ thống:

\begin{itemize}
    \item \textbf{Triển khai Ứng dụng Di động Native:} Đóng gói toàn bộ mô hình `.tflite` và luồng xử lý cục bộ vào một ứng dụng Android/iOS thực tế. Tiến hành đo benchmark chi tiết về tài nguyên phần cứng (CPU, RAM, Battery) trong nhiều điều kiện mạng khác nhau.
    
    \item \textbf{Tích hợp Federated Learning (Học liên kết):} Nâng cấp kiến trúc DCCL hiện tại lên một tầm cao mới bằng cách cho phép các thiết bị thực hiện tính toán Gradient cục bộ dựa trên lỗi dự đoán (loss), sau đó gửi mã hóa Gradient (thay vì dữ liệu) lên Cloud để cập nhật mô hình trung tâm một cách bảo mật.
    
    \item \textbf{Hỗ trợ Đồ thị Động (Dynamic Graphs):} Cải tiến thuật toán GNN để có khả năng cập nhật các thay đổi theo thời gian thực (thêm/bớt đỉnh, cạnh liên tục) mà không cần phải huấn luyện lại toàn bộ cấu trúc đồ thị từ đầu.
    
    \item \textbf{Kết hợp Đặc trưng Đa phương thức (Multi-modal Features):} Tích hợp thêm các phương pháp học sâu để trích xuất đặc trưng hình ảnh sản phẩm (CNN/ViT) bên cạnh đặc trưng văn bản, nhằm làm phong phú thêm chất lượng của các Vector nhúng (Item Embeddings) trên Cloud.
\end{itemize}
